    ```
    **처리 로직**:
    1. 1년 만기 예금 상품 조회 → 최고금리 TOP 3
    2. 1년 만기 적금 상품 조회 → 최고금리 TOP 3
    3. 예상 수익 계산 및 비교
    
    ```
    import requests
    
    def get_best_products(api_name, save_term, top_n=3):
        url = f"http://finlife.fss.or.kr/finlifeapi/{api_name}.json"
        params = {"auth": "YOUR_API_KEY", "topFinGrpNo": "020000", "pageNo": "1"}
        
        response = requests.get(url, params=params)
        data = response.json()['result']
        
        # 기간별 옵션 필터링
        products = []
        base_dict = {item['fin_prdt_cd']: item for item in data['baseList']}
        
        for option in data['optionList']:
            if option['save_trm'] == str(save_term):
                base = base_dict.get(option['fin_prdt_cd'])
                if base:
                    products.append({
                        'name': base['fin_prdt_nm'],
                        'company': base['kor_co_nm'],
                        'rate': float(option.get('intr_rate2', 0)),
                        'special_cond': base.get('spcl_cnd', '')
                    })
        
        # 금리 높은 순 정렬
        products.sort(key=lambda x: x['rate'], reverse=True)
        return products[:top_n]
    
    # 예금 상품 TOP 3
    deposits = get_best_products('depositProductsSearch', 12)
    print("=== 1년 만기 예금 추천 ===")
    for i, p in enumerate(deposits, 1):
        interest = 30_000_000 * p['rate'] / 100
        print(f"{i}. {p['company']} - {p['name']}")
        print(f"   금리: {p['rate']}% | 예상이자: {interest:,.0f}원\n")
    
    # 적금 상품 TOP 3
    savings = get_best_products('savingProductsSearch', 12)
    print("=== 1년 만기 적금 추천 ===")
    monthly = 70_000_000 / 12
    for i, p in enumerate(savings, 1):
        # 적금 이자 계산 (단리 가정)
        interest = monthly * 12 * (12+1) / 2 * p['rate'] / 100 / 12
        print(f"{i}. {p['company']} - {p['name']}")
        print(f"   금리: {p['rate']}% | 예상이자: {interest:,.0f}원\n")

금리가 가장 높은 적금 상품 정보를 가져오기 위해서는 금융감독원에서 제공하는 **적금 상품 조회 API(savingProductsSearch)**를 사용해야 합니다.

다음은 첨부해주신 프롬프트 지침에 따라 작성한 API 사용 방법과 예상 결과입니다.

1. 적절한 API 선택 및 이유
선택 API: savingProductsSearch

이유: 이 API는 사용자가 매월 일정 금액을 납입하는 적립식 저축 상품인 '적금' 데이터를 전문적으로 제공합니다. 기본 금리뿐만 아니라 우대 금리 정보를 포함하고 있어 금리 비교에 최적화되어 있습니다.

2. Python 요청 코드 예시
아래 코드는 은행권(020000) 적금 상품 중 **최고 우대금리(intr_rate2)**가 가장 높은 순서대로 상위 상품을 출력하는 예시입니다.

Python
import requests

# API 엔드포인트 및 파라미터 설정
url = "http://finlife.fss.or.kr/finlifeapi/savingProductsSearch.json"
params = {
    "auth": "YOUR_API_KEY",  # 발급받은 32자리 인증키
    "topFinGrpNo": "020000",   # 은행 기준 (저축은행은 030300)
    "pageNo": "1"
}

response = requests.get(url, params=params)
data = response.json()

# 상품 기본 정보와 옵션(금리) 정보 매핑
base_products = {item['fin_prdt_cd']: item for item in data['result']['baseList']}
options = data['result']['optionList']

savings_list = []
for option in options:
    product_info = base_products.get(option['fin_prdt_cd'])
    if product_info:
        savings_list.append({
            'company': product_info['kor_co_nm'],     # 금융회사명
            'name': product_info['fin_prdt_nm'],        # 상품명
            'base_rate': float(option.get('intr_rate') or 0),  # 기본 금리
            'max_rate': float(option.get('intr_rate2') or 0),  # 최고 우대금리
            'term': option['save_trm'],                 # 저축 기간
            'special_cond': product_info.get('spcl_cnd') # 우대 조건
        })

# 최고 우대금리 기준 내림차순 정렬
savings_list.sort(key=lambda x: x['max_rate'], reverse=True)

# 결과 출력
for i, s in enumerate(savings_list[:3], 1):
    print(f"{i}. {s['company']} - {s['name']} ({s['term']}개월)")
    print(f"   기본: {s['base_rate']}% | 최고: {s['max_rate']}%")
    print(f"   우대조건: {s['special_cond']}\n")
